FROM python:3.10-slim

# Install system dependencies and curl
RUN apt-get update && apt-get install -y curl && rm -rf /var/lib/apt/lists/*

# Install Ollama
RUN curl -fsSL https://ollama.com | sh

# Set up project directory
WORKDIR /code
COPY ./requirements.txt /code/requirements.txt
RUN pip install --no-cache-dir --upgrade -r /code/requirements.txt

# Copy application files
COPY . .

# Expose Gradio default port
EXPOSE 7860

# Shell script to start Ollama, pull models, then run the Python app
CMD ollama serve & sleep 5 && ollama pull llama3 && ollama pull nomic-embed-text && python app.py
